ingredients:
  list:
    - Whiskey
    - Angostura bitters
    - Agave Syrup
    - Orange
  qty:
    - 2 oz
    - 4 dashes 
    - 1/4 oz
    - swath

instructions:
  - Combine all ingredients in a mixing glass with ice.
  - Stir until well chilled.
  - Serve neat or on a rock.
  - Express the orange swath over the drink.
